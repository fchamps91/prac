# Write a Scala program to demonstrate an infinite while loop

object InfiniteWhileLoopDemo {
  def main(args: Array[String]): Unit = {
    var x = 1
    while (x < 5) {  // infinite loop since x never changes
      println("MSc CS")
    }
  }
}

# To Run 
 
InfiniteWhileLoopDemo.main(Array())
